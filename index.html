function getPolynom(...args) {
  const argsLength = args.length;
  if (argsLength === 0) return null;
  return (x) => {
    let result = 0;
    for (let i = 0; i < argsLength; i += 1) {
      result += args[i] * x ** (argsLength - 1 - i);
    }
    return result;
  };
}
